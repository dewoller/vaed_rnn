make audit data long

df_audit %>%
  mutate( row = row_number()) %>%
  select( row, starts_with('audit')
        , starts_with('hospital')) %>%
  gather( code, value, -row ) %>%
  separate( code, into=c('which', 'col' ), sep='_', n=2, extra='merge') %>%
  spread( which,value) %>% 
  filter( !is.na( hospital) & !is.na( audit )) %>%
  { . } -> df_audit_long
